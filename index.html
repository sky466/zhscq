def read_file(file_path):
    try:
        with open(file_path, 'r') as file:
            return file.read()
    except FileNotFoundError:
        print(f"文件 {file_path} 不存在")
    except Exception as e:
        print(f"读取文件时出错: {e}")

# 写入文件
def write_file(file_path, content):
    try:
        with open(file_path, 'w') as file:
            file.write(content)
